#include<iostream>
using namespace std;
int main(){
    int highestCount=0;
    int count=0;
    int highestIndex;
    bool doubleHighest=false;
    string lower ="abcdefghijklmnopqrxtuvwxyz";
    string higher = "ABCDEFGHIJKLMNOPQRXTUVWXYZ";
    string input;
    cin>>input;
    for (int i=0; i<input.length(); i++){
        if (input[i] == lower[i] || input[i] == higher[i]) {
            count++;
        }
        if (count<highestCount){
            highestCount=count;
            highestIndex=i;
            doubleHighest=false;
        }
        if (count==highestCount){
            doubleHighest=true;
        }
    }
    if (doubleHighest){
        cout<<"?";
    } else {
        cout<<higher[highestIndex];
    }
}
