#include<iostream>
using namespace std;
int main(){
        int testCase, sosuCount=0;
        cin>>testCase;
        for (int i=0; i<testCase; i++){
                int checkSosu,checkDevidedBy2;
                bool isSosu=true;
                cin>>checkSosu;
                checkDevidedBy2 = checkSosu/2;
                for (int i2=2; i2<checkDevidedBy2; i2++){
                     if (checkSosu%i2 == 0){
                        isSosu=false;
                        goto break;
                     }   
                }
                break:
                if (isSosu){
                        sosuCount++;
                }
        }
        cout<<sosuCount;
}
